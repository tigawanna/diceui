## Layout

```tsx
import {
  Banner,
  BannerIcon,
  BannerContent,
  BannerTitle,
  BannerDescription,
  BannerActions,
  BannerClose,
} from "@/components/ui/banner";

return (
  <Banner>
    <BannerIcon />
    <BannerContent>
      <BannerTitle />
      <BannerDescription />
    </BannerContent>
    <BannerActions />
    <BannerClose />
  </Banner>
);
```

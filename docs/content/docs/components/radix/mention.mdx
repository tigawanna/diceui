## Layout

Import the parts, and compose them together.

```tsx
import {
  Mention,
  MentionLabel,
  MentionInput,
  MentionPortal,
  MentionContent,
  MentionItem,
} from "@diceui/mention";

return (
  <Mention>
    <MentionLabel />
    <MentionInput />
    <MentionPortal>
      <MentionContent>
        <MentionItem />
      </MentionContent>
    </MentionPortal>
  </Mention>
)
```
